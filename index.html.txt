async function handleSignOut() {

// Handle auth state changes
onAuthStateChanged(auth, async (user) => {
  if (user) {
    currentUser = {
      uid: user.uid,
      displayName: user.displayName || "User",
      email: user.email || "",
      photoURL: user.photoURL || "",
    };
    await initUserSession(currentUser);
  } else {
    // Check for redirect result first (returning from signin page)
    try {
      const result = await getRedirectResult(auth);
      if (result?.user) {
        currentUser = {
          uid: result.user.uid,
          displayName: result.user.displayName || "User",
          email: result.user.email || "",
          photoURL: result.user.photoURL || "",
        };
        await initUserSession(currentUser);
      } else {
        renderApp();
      }
    } catch(e) {
      console.error("Redirect result error:", e);
      renderApp();
    }
  }
});

function F() { return currentFamily || {}; }
function allMeals() { return { ...MEALS, ...(F().customMeals||{}) }; }
function uniqueMeals() {
  const a = [];
  DAYS.forEach(d => MT.forEach(t => { if (F().weekPlan?.[d]?.[t]) a.push(F().weekPlan[d][t]); }));
  return [...new Set(a)];
}
function totalCost() { return uniqueMeals().reduce((s,m) => s + (allMeals()[m]?.cost||0), 0); }
function calcStart(mt,prep,cook) {
  const [h,m] = mt.split(":").map(Number);
  let sm = h*60+m-prep-cook; if(sm<0) sm+=1440;
  return `${String(Math.floor(sm/60)).padStart(2,"0")}:${String(sm%60).padStart(2,"0")}`;
}
function getDayNutr(day) {
  const t = {calories:0,protein:0,carbs:0,fat:0,fiber:0,fruits:0,vegetables:0,dairy:0,grains:0};
  MT.map(tp => F().weekPlan?.[day]?.[tp]).filter(Boolean).forEach(m => {
    const d = allMeals()[m]; if(!d) return;
    Object.keys(t).forEach(k => t[k] += (d[k]||0));
  });
  return t;
}
function getNutrAlerts(day) {
  const n=getDayNutr(day), tgt=DIET_T[F().activeDiet||"Standard"], a=[];
  if(n.fruits<tgt.fruits*.5) a.push({group:"Fruit",msg:`Only ${n.fruits} of ${tgt.fruits} fruit servings.`});
  if(n.vegetables<tgt.vegetables*.5) a.push({group:"Veggies",msg:`Only ${n.vegetables} of ${tgt.vegetables} veg servings.`});
  if(n.protein<tgt.protein*.5) a.push({group:"Protein",msg:`${n.protein}g of ${tgt.protein}g protein.`});
  return a;
}
function getNutrScore(day) {
  const n=getDayNutr(day), tgt=DIET_T[F().activeDiet||"Standard"];
  const keys=["protein","carbs","fat","fruits","vegetables"];
  return Math.round((keys.filter(k=>n[k]>=(tgt[k]||1)*.7).length/keys.length)*100);
}
function calcWeekScore() {
  const meals=Object.values(F().weekPlan||{}).flatMap(d=>Object.values(d)).filter(Boolean).length;
  const nutr=Math.round(DAYS.reduce((s,d)=>s+getNutrScore(d),0)/7);
  const ck=Object.values(userDoc?.cookedMeals||{}).filter(Boolean).length;
  const well=Object.values(userDoc?.wellnessLog||{}).filter(v=>v?.energy!==null&&v?.energy!==undefined).length;
  return {meals,nutrition:nutr,cooked:ck,wellness:well,total:meals*10+nutr+ck*20+well*15};
}

function buildAvatarEmoji(member) {
  const av = member.avatar || {};
  const toneObj = AV_TONES.find(t=>t.label===(av.tone||"Medium")) || AV_TONES[2];
  const baseEmoji = AV_BASE[member.avatarBase||"Woman"]?.emoji || "\u{1F469}";
  const baseWithTone = baseEmoji + toneObj.mod;
  const hairObj = AV_HAIR.find(h=>h.id===(av.hair||"straight"));
  const outfitObj = AV_OUTFITS.find(o=>o.id===(av.outfit||"casual"));
  const accObj = AV_ACCESSORIES.find(a=>a.id===(av.accessory||"none"));
  return {
    full: (accObj?.emoji||"") + (hairObj?.emoji||baseWithTone) + (outfitObj?.emoji||""),
    base: baseWithTone,
    acc: accObj?.emoji || "",
    hair: hairObj?.emoji || baseWithTone,
    outfit: outfitObj?.emoji || "",
  };
}
function renderAvatarBadge(member, size=32) {
  if (!member) return `<div style="width:${size}px;height:${size}px;border-radius:50%;background:var(--border);display:flex;align-items:center;justify-content:center;font-size:${Math.round(size*.44)}px">\u{1F464}</div>`;
  const av = buildAvatarEmoji(member);
  const color = member.color || "#69C9D0";
  const r = Math.round(size*.35);
  return `<div style="width:${size}px;height:${size}px;border-radius:${r}px;background:${color}22;border:2px solid ${color}44;display:flex;flex-direction:column;align-items:center;justify-content:center;flex-shrink:0;overflow:hidden;line-height:1" title="${member.name}"><span style="font-size:${Math.round(size*.2)}px">${av.acc}</span><span style="font-size:${Math.round(size*.38)}px">${av.base}</span><span style="font-size:${Math.round(size*.2)}px">${av.outfit}</span></div>`;
}

async function saveUserDoc(data) {
  if (!currentUser) return;
  try {
    setSyncStatus("syncing");
    await setDoc(doc(db,"users",currentUser.uid), { ...data, updatedAt:serverTimestamp() }, { merge:true });
    userDoc = { ...userDoc, ...data };
    setSyncStatus("online");
  } catch(e) {
    console.error("Save user error:", e);
    setSyncStatus("error");
  }
}
async function saveFamilyDoc(data) {
  if (!familyId) return;
  try {
    setSyncStatus("syncing");
    await setDoc(doc(db,"families",familyId), { ...data, updatedAt:serverTimestamp() }, { merge:true });
    currentFamily = { ...currentFamily, ...data };
    setSyncStatus("online");
  } catch(e) {
    console.error("Save family error:", e);
    setSyncStatus("error");
  }
}
function subscribeToFamily(fid) {
  const unsub = onSnapshot(doc(db,"families",fid), (snap) => {
    if (snap.exists()) {
      currentFamily = { id:fid, ...snap.data() };
      setSyncStatus("online");
      renderMain();
      renderHeader();
    }
  }, (e) => { setSyncStatus("error"); });
  unsubscribers.push(unsub);
}
function subscribeToUser(uid) {
  const unsub = onSnapshot(doc(db,"users",uid), (snap) => {
    if (snap.exists()) { userDoc = snap.data(); renderHeader(); }
  });
  unsubscribers.push(unsub);
}
function unsubscribeAll() {
  unsubscribers.forEach(u => u());
  unsubscribers = [];
}

// ── User Session ───────────────────────────────────────────
async function initUserSession(user) {
  const userRef = doc(db,"users",user.uid);
  const userSnap = await getDoc(userRef);
  if (!userSnap.exists()) {
    userDoc = {
      uid: user.uid,
      name: user.displayName || "User",
      email: user.email,
      photo: user.photoURL || "",
      xp: 0,
      gameBankMins: 30,
      wellnessLog: {},
      weekHistory: {},
      cookedMeals: {},
      checkedSteps: {},
      familyIds: [],
      createdAt: serverTimestamp(),
    };
    await setDoc(userRef, userDoc);
  } else {
    userDoc = userSnap.data();
  }
  subscribeToUser(user.uid);
  if (userDoc.familyIds?.length > 0) {
    familyId = userDoc.familyIds[0];
    subscribeToFamily(familyId);
    const famSnap = await getDoc(doc(db,"families",familyId));
    if (famSnap.exists()) {
      currentFamily = { id:familyId, ...famSnap.data() };
    }
  }
  renderApp();
}

async function createFamily(name) {
  if (!currentUser) { showGetActiveModal(); return; }
  try {
    const inviteCode = genInviteCode();
    const defaultWeekPlan = {};
    DAYS.forEach(d => { defaultWeekPlan[d] = {Breakfast:null,Lunch:null,Dinner:null,Snack:null}; });
    const famData = {
      name: name || `${currentUser.displayName?.split(" ")[0]}'s Family`,
      createdBy: currentUser.uid,
      inviteCode,
      memberIds: [currentUser.uid],
      members: [{
        id: currentUser.uid,
        name: currentUser.displayName || "Parent",
        emoji: "👩",
        color: "#69C9D0",
        diet: "Standard",
        goalType: "maintain",
        picky: false,
        avoid: [],
        loves: [],
        moods: [],
        games: [],
        isAdmin: true,
      }],
      weekPlan: defaultWeekPlan,
      prefDays: {Mon:null,Tue:null,Wed:null,Thu:null,Fri:null,Sat:null,Sun:null},
      messages: [],
      videos: [],
      challenges: [],
      activeDiet: "Standard",
      size: 1,
      weekHistory: [],
      customMeals: {},
      createdAt: serverTimestamp(),
    };
    const famRef = await addDoc(collection(db,"families"), famData);
    familyId = famRef.id;
    currentFamily = { id:familyId, ...famData };
    await saveUserDoc({ familyIds: arrayUnion(familyId) });
    subscribeToFamily(familyId);
    renderApp();
    showToast(`Family created! Invite code: ${inviteCode}`, "success");
  } catch(e) {
    console.error("Create family error:", e);
    showToast("Error creating family.", "error");
  }
}

async function joinFamily(code) {
  if (!currentUser) { showGetActiveModal(); return; }
  try {
    const q = query(collection(db,"families"), where("inviteCode","==",code.toUpperCase()));
    const snap = await getDocs(q);
    if (snap.empty) { showToast("Invalid invite code.", "error"); return; }
    const famDoc = snap.docs[0];
    familyId = famDoc.id;
    const famData = famDoc.data();
    if (famData.memberIds?.includes(currentUser.uid)) {
      showToast("You're already in this family!", "warn");
      currentFamily = { id:familyId, ...famData };
      subscribeToFamily(familyId);
      renderApp();
      return;
    }
    const newMember = {
      id: currentUser.uid,
      name: currentUser.displayName || "Member",
      emoji: "🧑",
      color: "#a78bfa",
      diet: "Standard",
      goalType: "maintain",
      picky: false,
      avoid: [],
      loves: [],
      moods: [],
      games: [],
    };
    await updateDoc(doc(db,"families",familyId), {
      memberIds: arrayUnion(currentUser.uid),
      members: arrayUnion(newMember),
      size: (famData.size||1) + 1,
    });
    await saveUserDoc({ familyIds: arrayUnion(familyId) });
    currentFamily = { id:familyId, ...famData };
    subscribeToFamily(familyId);
    renderApp();
    showToast(`Joined ${famData.name}!`, "success");
  } catch(e) {
    showToast("Error joining family.", "error");
  }
}

    emoji: myMember?.emoji || "🧑",
    text,
    time: new Date().toLocaleTimeString([], {hour:'2-digit',minute:'2-digit'}),

function initMemory() {
  const items = [
    {emoji:"🥦",pair:"a"},{emoji:"🥦",pair:"a"},
    {emoji:"🍎",pair:"b"},{emoji:"🍎",pair:"b"},
    {emoji:"🧀",pair:"c"},{emoji:"🧀",pair:"c"},
    {emoji:"🥕",pair:"d"},{emoji:"🥕",pair:"d"},
    {emoji:"🍋",pair:"e"},{emoji:"🍋",pair:"e"},
    {emoji:"🥑",pair:"f"},{emoji:"🥑",pair:"f"},
  ];
  UI.memCards = items.sort(() => Math.random() - 0.5);
  UI.memFlipped = [];
  UI.memMatched = [];
  UI.memLocked = false;
  renderMain();
}

function tapMemCard(idx) {
  if (!UI.memCards) return;
  if (UI.memLocked) return;
  if ((UI.memMatched||[]).includes(idx)) return;
  if ((UI.memFlipped||[]).includes(idx)) return;
  if ((UI.memFlipped||[]).length >= 2) return;
  UI.memFlipped = [...(UI.memFlipped||[]), idx];
  renderMain();
  if (UI.memFlipped.length === 2) {
    const [a, b] = UI.memFlipped;
    if (UI.memCards[a].pair === UI.memCards[b].pair) {
      UI.memMatched = [...(UI.memMatched||[]), a, b];
      UI.memFlipped = [];
      if (UI.memMatched.length === UI.memCards.length) { addXP(20); showToast("+20 XP! You matched everything! 🎉", "success"); }
      renderMain();
    } else {
      UI.memLocked = true;
      setTimeout(() => { UI.memFlipped = []; UI.memLocked = false; renderMain(); }, 900);
    }
  }
}

async function sendAIMsg() {
  const input = document.getElementById("ai-input");
  const text = input?.value?.trim();
  if (!text || UI.aiLoading) return;
  UI.aiChat.push({role:"user",text});
  UI.aiInput = "";
  UI.aiLoading = true;
  if (input) input.value = "";
  renderMain();
  const reply = await callAI(UI.aiChat.map(m=>({role:m.role,content:m.text})), "You are NeuroMeal AI, a helpful meal planning assistant for AuDHD families. Be warm, practical, and encouraging. Keep responses concise.");
  UI.aiLoading = false;
  UI.aiChat.push({role:"assistant",text:reply});
  renderMain();
}

async function callAI(messages, system="You are a helpful assistant.") {
  try {
    const response = await fetch("/api/ai", {
      method: "POST",
      headers: {"Content-Type":"application/json"},
      body: JSON.stringify({messages, system}),
    });
    const data = await response.json();
    return data.content?.[0]?.text || "Sorry, I couldn't respond right now.";
  } catch(e) {
    return "AI is taking a break. Try again in a moment!";
  }
}

function initMatch() {
  const items = [
    {emoji:"🥦",pair:"broccoli"},{emoji:"🥦",pair:"broccoli"},
    {emoji:"🍎",pair:"apple"},{emoji:"🍎",pair:"apple"},
    {emoji:"🧀",pair:"cheese"},{emoji:"🧀",pair:"cheese"},
    {emoji:"🥕",pair:"carrot"},{emoji:"🥕",pair:"carrot"},
  ];
  UI.matchCards = items.sort(() => Math.random() - 0.5);
  UI.matchFlipped = [];
  UI.matchMatched = [];
  UI.matchScore = 0;
}

function flipCard(idx) {
  if (UI.matchMatched.includes(idx) || UI.matchFlipped.includes(idx)) return;
  if (UI.matchFlipped.length >= 2) { UI.matchFlipped = []; renderMain(); return; }
  UI.matchFlipped = [...UI.matchFlipped, idx];
  renderMain();
  if (UI.matchFlipped.length === 2) {
    const [a,b] = UI.matchFlipped;
    if (UI.matchCards[a].pair === UI.matchCards[b].pair) {
      UI.matchMatched = [...UI.matchMatched, a, b];
      UI.matchFlipped = [];
      UI.matchScore++;
      if (UI.matchMatched.length === UI.matchCards.length) { addXP(20); showToast("+20 XP! All matched! 🎉", "success"); }
      renderMain();
    } else {
      setTimeout(() => { UI.matchFlipped = []; renderMain(); }, 900);
    }
  }
}
